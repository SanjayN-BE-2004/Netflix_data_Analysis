text = " ".join(desc for desc in df['description'].dropna())

wordcloud = WordCloud(
    width=800, 
    height=400, 
    background_color='black',  
    colormap='viridis',       
    max_words=100             
).generate(text)

plt.figure(figsize=(12, 6))
plt.imshow(wordcloud, interpolation='bilinear')
plt.axis("off")
plt.show()
